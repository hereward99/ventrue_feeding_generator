        const cats = {
            Music: {r: ['Pop Star', 'Rock Musician', 'Hip-Hop Artist', 'Opera Singer', 'DJ'], l: ['Recording Studio', 'Concert Venue', 'Nightclub VIP', 'Hotel Suite']},

            Crime: {r: ['Crime Boss', 'Financier', 'Dealer', 'Money Launderer'], l: ['Casino Backroom', 'Underground Club', 'Warehouse Office', 'Strip Club VIP']},
            SocialMedia: {r: ['Influencer', 'YouTube Star', 'TikTok Celebrity', 'Streamer'], l: ['Creation Studio', 'Influencer Event', 'Creator House', 'Pop-up Store']}
        };
        
        const catNames = {
            Music: 'Music',
            FilmTV: 'Film and TV',
            Comedy: 'Comedy',
            Politics: 'Politics',
            BizTech: 'Business and Tech',
            Sports: 'Sports',
            Culture: 'Culture and Academia',

        };
        
        const names = {
            english: {f: ['James', 'Mary', 'Robert', 'Patricia', 'John', 'Jennifer', 'Michael', 'Linda', 'David', 'Elizabeth'], l: ['Smith', 'Johnson', 'Williams', 'Brown', 'Jones', 'Garcia', 'Miller', 'Davis']},
            italian: {f: ['Giuseppe', 'Maria', 'Antonio', 'Anna', 'Francesco', 'Rosa'], l: ['Rossi', 'Russo', 'Ferrari', 'Esposito', 'Bianchi']},
            french: {f: ['Jean', 'Marie', 'Michel', 'Pierre', 'Andre'], l: ['Martin', 'Bernard', 'Thomas', 'Petit', 'Robert']},

            spanish: {f: ['Antonio', 'Maria', 'Jose', 'Carmen'], l: ['Garcia', 'Gonzalez', 'Rodriguez', 'Fernandez']},
            nordic: {f: ['Erik', 'Anna', 'Lars', 'Emma'], l: ['Andersson', 'Johansson', 'Karlsson', 'Nilsson']},

            middleeastern: {f: ['Ahmed', 'Fatima', 'Mohammad', 'Aisha'], l: ['Al-Ahmad', 'Mahmoud', 'Ibrahim', 'Mansour']}
        };
        
        const cityMap = {'london':'english','new york':'english','los angeles':'english','paris':'french','berlin':'german','rome':'italian','madrid':'spanish','stockholm':'nordic','moscow':'slavic','tokyo':'eastasian','dubai':'middleeastern'};
        
        const tiers = [{t:'A',d:'Internationally known household name'},{t:'B',d:'Nationally famous'},{t:'C',d:'TV cast, regionally known'},{t:'D',d:'Reality TV stars'},{t:'Z',d:'Influencers, niche personalities'}];
        const sec = [{l:'Minimal',i:'🟢',d:'Personal assistant only'},{l:'Light',i:'🟡',d:'Small security'},{l:'Moderate',i:'🟠',d:'Bodyguards'},{l:'Heavy',i:'🔴',d:'Multiple layers'}];
        
        let state = {city:'',target:null,sel:{}};
        Object.keys(cats).forEach(function(k){
            state.sel[k] = true;
        });
        state.sel.Associates = true;
        
        function r(a){return a[Math.floor(Math.random()*a.length)];}
        function gn(c){const reg=cityMap[c.toLowerCase().trim()]||'english';const n=names[reg];return r(n.f)+' '+r(n.l);}
        
        function getWeightedTier(){
            const pool=[];
            for(let i=0;i<1;i++)pool.push('A');
            for(let i=0;i<2;i++)pool.push('B');
            for(let i=0;i<5;i++)pool.push('C');
            for(let i=0;i<6;i++)pool.push('D');
            for(let i=0;i<6;i++)pool.push('Z');
            const tier=r(pool);
            return tiers.find(t=>t.t===tier);
        }
        
        function getDifficultyForTier(tier){
            const ranges={'A':[3,4,5],'B':[2,3,4,5],'C':[1,2,3,4],'D':[1,2,3],'Z':[0,1,2,3]};
            return r(ranges[tier]);
        }
        
        function ga(p,t){
            const rel=['Spouse of','Partner of','Parent of','Child of','Sibling of','Manager of'];
            const roles=['Socialite','Investor','Philanthropist','Designer','Restaurant Owner'];
            const tm={'A':'C','B':'D','C':'Z','D':'Z','Z':'Z'};
            return{rel:r(rel),role:r(roles),pc:p,fame:tiers.find(f=>f.t===tm[t])};
        }
        
        function toggleCat(k){
            state.sel[k] = !state.sel[k];
            render();
        }
        
        function updateCity(val){
            state.city = val;
        }
        
        function checkEnter(e){
            if(e.key === 'Enter') gen();
        }
        
        function gen(){
            if(!state.city.trim())return;
            const cs=Object.keys(cats).filter(k=>state.sel[k]).map(k=>({cat:k,r:cats[k].r,l:cats[k].l}));
            if(state.sel.Associates)cs.push({cat:'Associates'});
            if(!cs.length){alert('Select at least one category');return;}
            const c=r(cs);
            let t;
            const f=getWeightedTier();
            const diff=getDifficultyForTier(f.t);
            if(c.cat==='Associates'){
                const nc=Object.keys(cats).map(k=>({cat:k,r:cats[k].r}));
                const pc=r(nc);
                const pr=r(pc.r);
                const pn=gn(state.city);
                const ai=ga(pn,f.t);
                const locs=['Private Residence','Spa','Shopping District','Hotel Lounge','Members Club','Gallery','Charity Event'];
                t={name:gn(state.city),role:ai.role,cat:'Associates',loc:r(locs)+' - '+state.city,sec:r(sec),fame:ai.fame,isA:true,rel:ai.rel,pc:ai.pc,pr:pr,diff:getDifficultyForTier(ai.fame.t)};
            }else{
                t={name:gn(state.city),role:r(c.r),cat:c.cat,loc:r(c.l)+' - '+state.city,sec:r(sec),fame:f,isA:false,diff:diff};
            }
            const hooks=['Recently in scandal','Major achievement','Secret business deal','Private charity event','Meeting power brokers','Recovering from controversy','Launching venture','Power struggle'];
            const comps=['Heavy media nearby','Rival faction interested','Extra security','Many witnesses','Bodyguard is ghoul','Mystical wards','Target suspicious','Police investigation'];
            t.hook=r(hooks);
            t.comp=r(comps);
            state.target=t;
            render();
        }
        
        function render(){
            var app=document.getElementById('app');
            var allOff=Object.values(state.sel).every(function(v){return !v;});
            var html = '';
            html += '<div class="container"><div class="header"><h1 class="title">Rosas Ventrue Feeding Generator</h1><p class="subtitle">Find worthy prey for your discriminating Ventrue</p></div>';
            html += '<div class="card"><h3 class="card-title">Target Categories</h3><p class="card-text">Select which types of targets are appropriate for your current scene:</p><div class="checkbox-grid">';
            Object.keys(cats).forEach(function(k){
                var chk = state.sel[k] ? 'checked' : '';
                var cls = state.sel[k] ? 'checkbox-text-active' : 'checkbox-text-inactive';
                html += '<label class="checkbox-label"><input type="checkbox" class="checkbox" data-cat="'+k+'" '+chk+'><span class="checkbox-text '+cls+'">'+catNames[k]+'</span></label>';
            });
            var chkA = state.sel.Associates ? 'checked' : '';
            var clsA = state.sel.Associates ? 'checkbox-text-active' : 'checkbox-text-inactive';
            html += '<label class="checkbox-label"><input type="checkbox" class="checkbox" data-cat="Associates" '+chkA+'><span class="checkbox-text '+clsA+'">Associates</span></label>';
            html += '</div>';
            if(allOff) html += '<div class="warning">Warning: No target categories selected</div>';
            html += '</div>';
            var disabled = '';
            if(!state.city.trim() || allOff){
                disabled = 'disabled';
            }
            html += '<div class="card"><div class="input-group"><div class="input-wrapper"><label class="label">Enter City Name</label><input type="text" id="cityInput" class="input" value="'+state.city+'" placeholder="e.g., New York, London, Tokyo..."></div>';
            html += '<button class="button" id="genBtn" '+disabled+'>Find Target</button></div></div>';
            if(state.target){
                var t=state.target;
                html += '<div class="card"><div class="target-grid"><div><div class="target-name">'+t.name+'</div>';
                if(t.isA) html += '<div class="associate-box"><div class="associate-text"><strong>'+t.rel+'</strong> '+t.pc+'</div><div class="associate-subtext">('+t.pr+')</div></div>';
                html += '<div class="info-row"><span class="info-label">Role:</span> '+t.role+'</div>';
                html += '<div class="info-row"><span class="info-label">Location:</span> '+t.loc+'</div>';
                html += '<div class="info-row"><span class="info-label">Fame Tier:</span> '+t.fame.t+'-List</div>';
                html += '<div class="fame-box">'+t.fame.d+'</div>';
                html += '<div class="info-row"><span class="info-label">Security:</span> '+t.sec.i+' '+t.sec.l+' ('+t.sec.d+')</div>';
                var stars = '<div class="info-row"><span class="info-label">Difficulty:</span> <span class="stars">';
                for(var i=0;i<t.diff;i++) stars+='★';
                for(var i=t.diff;i<5;i++) stars+='☆';
                stars+='</span></div>';
                html += stars;
                html += '</div><div><h3 class="story-section-title">Story Elements</h3>';
                html += '<div class="story-box"><h4 class="story-title">Current Situation</h4><p class="story-text">'+t.hook+'</p></div>';
                html += '<div class="complication-box"><h4 class="complication-title">Potential Complication</h4><p class="complication-text">'+t.comp+'</p></div>';
                html += '<div class="opportunity-box"><h4 class="opportunity-title">Feeding Opportunity</h4><p class="opportunity-text">Target likely alone between <strong>'+(Math.floor(Math.random()*4)+21)+':00-'+(Math.floor(Math.random()*4)+1)+':00</strong> when security is lightest.</p></div>';
                html += '</div></div></div>';
            }
            html += '<div class="footer"><p>Remember: The Masquerade must be maintained at all costs.</p></div></div>';
            app.innerHTML = html;
            
            document.querySelectorAll('input[type="checkbox"]').forEach(function(cb){
                cb.addEventListener('click',function(){
                    var cat = this.getAttribute('data-cat');
                    state.sel[cat] = !state.sel[cat];
                    render();
                });
            });
            
            var cityInput = document.getElementById('cityInput');
            if(cityInput){
                cityInput.addEventListener('input',function(){
                    state.city = this.value;
                    var genBtn = document.getElementById('genBtn');

                        var allOff = Object.values(state.sel).every(function(v){return !v;});
                        if(!state.city.trim() || allOff){
                            genBtn.setAttribute('disabled', 'disabled');
                        } else {
                            genBtn.removeAttribute('disabled');

                });

                    if(e.key === 'Enter'){
                        gen();
                    }
                });
            }
            
            var genBtn = document.getElementById('genBtn');
            if(genBtn){
                genBtn.addEventListener('click', function(){
                    gen();
                });
            }
        }

            render();
        };
